Event

  • eventType : String
    Type of change that is done to the entity
    Enum:
    CREATE
    DELETE
    UPDATE
  • entityType : String
    Entity that is changed.
    Enum:
    duty
    timesheet
    booking
    grade
    agency
    reason_for_vacancy
    contract_status
    staff_group
    unit
  • entityId : String
    Trust identifier of the changed entity
  • timeStamp : Date
    Timestamp of the event
